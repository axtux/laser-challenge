root level mandatory with attributes :
  name mandatory
  map mandatory
  difficulty default "medium"
  id mandatory
blocks mandatory within level
block optional within blocks with attributes :
  type mandatory
  fixedposition default false
  fixedorientation default false
  orientation default UP (first from orienations.xml)
position optional attribute within block with attributes :
  x default -1
  y default -1
  (if blocks out of map go to inventary)
